# 重载自加/自减运算符
```cpp
class Text{
private:
    int data;
public:
    //前加
    Text operator ++ (){
        ++data;
        return *this;
    }
    //后加
    Text operator ++ (int){
        Text now = *this;
        ++data;
        return now;
    }
    //前减
    Text operator -- (){
        --data;
        return *this;
    }
    //后减
    Text operator -- (int){
        Text now = *this;
        --data;
        return now;
    }
}

```
